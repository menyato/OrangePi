# requirements_orangepi.txt
# OrangePi Zero 2W — glove hub + voice client (gesture engine, STT, TTS, camera)
# ──────────────────────────────────────────────────────────────────────────────
# System packages first (run once):
#   sudo apt update
#   sudo apt install -y espeak-ng alsa-utils portaudio19-dev libsndfile1 \
#                       python3-pip python3-dev build-essential python3-opencv
#
# Then install Python packages:
#   pip3 install --break-system-packages -r requirements_orangepi.txt
#
# NOTE on OpenCV: always install via apt (python3-opencv) as shown above — do
# NOT use pip on a 512 MB ARM board (builds from source, OOMs).  Only fall back
# to the pip headless wheel if apt is unavailable:
#   # opencv-python-headless==4.10.0.84
# ──────────────────────────────────────────────────────────────────────────────

# ── Serial / glove hardware ───────────────────────────────────────────────────
# UART driver used by glove_controller.py
pyserial==3.5

# ── Speech-to-text ────────────────────────────────────────────────────────────
# faster-whisper: CTranslate2-backed Whisper, runs int8 on CPU — no PyTorch
faster-whisper==1.1.1

# ── Voice activity detection ──────────────────────────────────────────────────
# Lightweight C extension; replaces the heavier silero-vad + torch stack
webrtcvad==2.0.10

# ── Audio I/O ─────────────────────────────────────────────────────────────────
# PortAudio Python bindings (mic capture, earcon playback)
sounddevice==0.5.1

# ── Speech recognition ────────────────────────────────────────────────────────
# Used by gesture_hub/voice.py for the voice command listener
SpeechRecognition>=3.10

# ── Numerics ──────────────────────────────────────────────────────────────────
# Required by faster-whisper, cv2 keyframe extraction, and audio processing
numpy>=1.24,<3.0

# ── HTTP ──────────────────────────────────────────────────────────────────────
# Used by ServerLink (net/client.py) for the money / OCR server connection
requests

# ── Environmental Awareness (Gemini vision API) ───────────────────────────────
# google-genai is the official Google AI Python SDK (replaces google-generativeai)
# Used by features/env_awareness.py to send keyframes to Gemini 2.5 Flash.
# Requires a GEMINI_API_KEY — see README or add to ~/.env on the Pi.
google-genai>=1.0.0

# ── LiDAR (MS200) + SLAM ──────────────────────────────────────────────────────
# pyserial already listed above — used by both GloveController and MS200Adapter.
#
# scipy: medfilt (lidar_adapter.py), least_squares + KDTree (slam_engine.py).
# Install the pre-built ARMv8 wheel from piwheels — do NOT build from source
# on a 512 MB board (it will OOM or take 30+ minutes):
#   pip3 install --break-system-packages \
#       --extra-index-url https://www.piwheels.org/simple scipy
scipy>=1.11

# OPTIONAL — GPU acceleration via CuPy (ICP + ScanContext distance).
# OrangePi Zero 2W has no CUDA; slam_engine.py falls back to scipy/CPU
# automatically when cupy is not importable.  Only install if you attach
# a CUDA-capable co-processor or use a different board:
# cupy-cuda12x

# ── OPTIONAL: noise cancellation ─────────────────────────────────────────────
# Runs a spectral-gating filter before Whisper transcription.
# Improves accuracy in noisy environments at the cost of ~100 ms extra latency.
# Uncomment to enable:
# noisereduce
